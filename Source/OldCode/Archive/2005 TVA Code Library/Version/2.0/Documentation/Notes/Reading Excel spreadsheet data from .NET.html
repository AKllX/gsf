Subject: Reading Excel Spreadsheet Data from .NET...

 

Sample connect string for OLEDB.NET:

 

Provider=Microsoft.Jet.OleDb.4.0;Data Source=C:\NES.xls;Extended Properties=Excel 8.0

 

Uses named ranges for table names or the worksheets can be directly referenced as a Table (for example table name for default “Sheet1” is “Sheet1$”)

 

First row in spread sheet defines column names…  Example:

 

Imports System.Data.OleDb

 

Module Module1

 

    Sub Main()

 

        Dim cnn As OleDbConnection

        Dim cmd As OleDbCommand

        Dim rst As OleDbDataReader

 

        Try

            cnn = New OleDbConnection("Provider=Microsoft.Jet.OleDb.4.0;Data Source=C:\NES2.xls;Extended Properties=Excel 8.0")

            cnn.Open()

            cmd = New OleDbCommand("ANALOG$", cnn)

            cmd.CommandType = CommandType.TableDirect

            rst = cmd.ExecuteReader()

 

            Do While rst.Read()

                Console.WriteLine("Row: " & rst("Substation") & rst("EntityName"))

            Loop

 

            rst.Close()

            cnn.Close()

        Catch ex As Exception

            Console.WriteLine("Connect Error: " & ex.Message)

        End Try

 

        Do While Console.ReadLine()

 

        Loop

 

    End Sub

 

End Module
