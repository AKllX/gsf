Subject: SQL Server Holiday Date Calculation Functions

The new Real Time Pricing database calculates prices for some contracts based on holidays – in order to keep the end users from having to manually enter the observed holiday dates for each year I though it would be better to calculate the holidays.  I searched the web and found algorithms for holiday date calculations from several different sources and languages and translated them into the following SQL server functions.  I’ve hand checked the results from 1988 to 2020 (see table below functions).  I only included holidays for which I thought there may be a day off – others could be easily added…

 

The only complicated one was the Easter Sunday calculation since the actual date is based on a lunar calendar, but I tested the function’s algorithm results against the U.S. Naval Observatory’s Easter data calculator (http://aa.usno.navy.mil/data/docs/easter.html) with accurate results.

 

All functions take a year INT argument and return a DATETIME of the holiday…

 

Holiday Functions:

 

CREATE FUNCTION dbo.HolidayFormula_NewYearsDay (@Year INT) 

RETURNS DATETIME  AS 

BEGIN

 

      RETURN CAST(CAST(@Year AS CHAR(4)) + '-01-01' AS DATETIME)

 

END

 

CREATE FUNCTION dbo.HolidayFormula_NewYearsDayObserved (@Year INT)

RETURNS DATETIME  AS 

BEGIN

 

/*

      New Year's Day observance seems fairly simple at first until we remember the wrinkle in the calculation that if the

      holiday falls on a Saturday, then the day off is the preceding Friday, December 31st -- New Year's Eve.

*/

 

      DECLARE @Holiday DATETIME

      DECLARE @WeekDay INT

 

      SET @WeekDay = DATEPART(dw, dbo.HolidayFormula_NewYearsDay(@Year))

 

      SET @Holiday =

      CASE

            WHEN @WeekDay = 1 THEN  /* Sunday */

                  CAST(CAST(@Year AS CHAR(4)) + '-01-02' AS DATETIME)

            WHEN @WeekDay <= 6 THEN /* Monday - Friday */

                  CAST(CAST(@Year AS CHAR(4)) + '-01-01' AS DATETIME)

            ELSE                    /* Saturday */

                  CAST(CAST((@Year - 1) AS CHAR(4)) + '-12-31' AS DATETIME)

      END

 

      RETURN @Holiday

 

END

CREATE FUNCTION dbo.HolidayFormula_EasterSunday (@Year INT)

RETURNS DATETIME  AS 

BEGIN

 

/*

      This function can calculate Easter Sunday from 1583 (beginning of Gregorian calendar) to 2499 (end of correction values).

*/

      DECLARE @Correction INT

      DECLARE @Month INT

      DECLARE @Day INT

 

      SET @Correction =

      CASE

            WHEN @Year < 1700 THEN

                  4

            WHEN @Year < 1800 THEN

                  5

            WHEN @Year < 1900 THEN

                  6

            WHEN @Year < 2100 THEN

                  0

            WHEN @Year < 2200 THEN

                  1

            WHEN @Year < 2300 THEN

                  2

            WHEN @Year < 2500 THEN

                  3

      END

 

      SET @Day = (19 * (@Year % 19) + 24) % 30

      SET @Day = 22 + @Day + ((2 * (@Year % 4) + 4 * (@Year % 7) + 6 * @Day + 5 + @Correction) % 7)

   

      /* Jump to next month */

      IF @Day > 31

      BEGIN

            SET @Month = 4

            SET @Day = @Day - 31

      END

 

      ELSE

            SET @Month = 3

 

      RETURN CAST(CAST(@Year AS CHAR(4)) + '-' + CAST(@Month AS VARCHAR(2)) + '-' + CAST(@Day AS VARCHAR(2)) AS DATETIME)

 

END

 

 

CREATE FUNCTION dbo.HolidayFormula_AshWednesday (@Year INT)

RETURNS DATETIME  AS 

BEGIN

 

/*

      Ash Wednesday is 46 days before Easter Sunday.

*/

      RETURN DATEADD(day, -46, dbo.HolidayFormula_EasterSunday(@Year))

 

END

 

 

CREATE FUNCTION dbo.HolidayFormula_GoodFridayObserved (@Year INT)

RETURNS DATETIME  AS 

BEGIN

 

/*

      Good Friday, as observed by the New York Stock Exchange, is 2 days before Easter Sunday.

*/

      RETURN DATEADD(day, -2, dbo.HolidayFormula_EasterSunday(@Year))

 

END

CREATE FUNCTION dbo.HolidayFormula_IndependenceDayObserved (@Year INT)

RETURNS DATETIME  AS 

BEGIN

 

/*

      Independence Day, July 4th, is another easy day to calculate, since the only effort required is to see if the holiday falls

      on the weekend. If it does, then the observance rule should be applied to determine if businesses will observe the holiday

      on either the preceding Friday or the following Monday.

*/

 

      DECLARE @Holiday DATETIME

      DECLARE @WeekDay INT

 

      SET @WeekDay = DATEPART(dw, dbo.HolidayFormula_IndependenceDay(@Year))

 

      SET @Holiday =

      CASE

            WHEN @WeekDay = 1 THEN  /* Sunday */

                  CAST(CAST(@Year AS CHAR(4)) + '-07-05' AS DATETIME)

            WHEN @WeekDay <= 6 THEN /* Monday - Friday */

                  CAST(CAST(@Year AS CHAR(4)) + '-07-04' AS DATETIME)

            ELSE                    /* Saturday */

                  CAST(CAST(@Year AS CHAR(4)) + '-07-03' AS DATETIME)

      END

 

      RETURN @Holiday

 

END

CREATE FUNCTION dbo.HolidayFormula_ChristmasDayObserved (@Year INT)

RETURNS DATETIME  AS 

BEGIN

 

/*

      Christmas Day is December 25th, and if the 25th falls on a Saturday, Christmas is observed by businesses on Christmas Eve,

      December 24th. If December 25th falls on a Sunday, then it will be observed by businesses in the United States on the 26th.

*/

 

      DECLARE @Holiday DATETIME

      DECLARE @WeekDay INT

 

      SET @WeekDay = DATEPART(dw, dbo.HolidayFormula_ChristmasDay(@Year))

 

      SET @Holiday =

      CASE

            WHEN @WeekDay = 1 THEN  /* Sunday */

                  CAST(CAST(@Year AS CHAR(4)) + '-12-26' AS DATETIME)

            WHEN @WeekDay <= 6 THEN /* Monday - Friday */

                  CAST(CAST(@Year AS CHAR(4)) + '-12-25' AS DATETIME)

            ELSE                    /* Saturday */

                  CAST(CAST(@Year AS CHAR(4)) + '-12-24' AS DATETIME)

      END

 

      RETURN @Holiday

 

END
